144 inline struct timespec timespec_add(struct timespec time1, struct timespec time2)
145 {
146  struct timespec result;
147 
148  if ((time1.tv_nsec + time2.tv_nsec) >= g_kNsPerSec)
149  {
150  result.tv_sec = time1.tv_sec + time2.tv_sec + 1;
151  result.tv_nsec = time1.tv_nsec + time2.tv_nsec - g_kNsPerSec;
152  }
153  else
154  {
155  result.tv_sec = time1.tv_sec + time2.tv_sec;
156  result.tv_nsec = time1.tv_nsec + time2.tv_nsec;
157  }
158 
159  return result;
160 }
