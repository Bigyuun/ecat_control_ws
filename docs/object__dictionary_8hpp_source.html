200/* From CiA402, - State coding
201 Statusword | PDS FSA state
202xxxx xxxx x0xx 0000 | Not ready to switch on
203xxxx xxxx x1xx 0000 | Switch on disabled
204xxxx xxxx x01x 0001 | Ready to switch on
205xxxx xxxx x01x 0011 | Switched on
206xxxx xxxx x01x 0111 | Operation enabled
207xxxx xxxx x00x 0111 | Quick stop active
208xxxx xxxx x0xx 1111 | Fault reaction active
209xxxx xxxx x0xx 1000 | Fault
210*/
211#define SM_FSAFROMSTATUSWORD(SW) (SW & 0x006f)
212#define SM_NOT_READY_TO_SWITCH_ON 0b00000000
213#define SM_NOT_READY_TO_SWITCH_ON_2 0b00100000
214#define SM_SWITCH_ON_DISABLED 0b01000000
215#define SM_SWITCH_ON_DISABLED_2 0b01100000
216#define SM_READY_TO_SWITCH_ON 0b00100001
217#define SM_SWITCHED_ON 0b00100011
218#define SM_OPERATION_ENABLED 0b00100111
219#define SM_QUICK_STOP_ACTIVE 0b00000111
